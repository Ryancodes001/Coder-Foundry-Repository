Enter numbers below to see them Fizz and Buzz!



Look at those FizzBuzzes!

    $(document).ready(function() {
$('#fizzButton').click(function(event) {
     event.preventDefault();
     $('#fizzOutput').empty();
	var input1 = $('#fizzInput1').val();
	var input2 = $('#fizzInput2').val();
       
if  (((input1 > 0) && (input1 <= 100)) && ((input2 > 0) && (input2 <= 100))) {
for (var i=1; i <= 100; i++){
   if (i % input1===0 && i % input2 ===0) {
       $('#fizzOutput').append("FizzBuzz" + "
");
         
        }else if (i % input1 === 0){
            $('#fizzOutput').append("Fizz" + "
");
        }
       
        else if (i % input2 === 0){
            $('#fizzOutput').append("Buzz" + "
");
        }
        
        else{
            $('#fizzOutput').append(i + "
");
        }
     }
    }
    else {
        $('#fizzOutput').append(" Please enter 2 numbers between 1-100")
    }
});
});
